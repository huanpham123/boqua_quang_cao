// SAFE_SCRIPT_v1
(function(){
  'use strict';
  try {
    // Thử tìm thẻ meta redirect
    const meta = document.querySelector('meta[http-equiv="refresh"]');
    if (meta) {
      const match = (meta.content || '').match(/url=(.*)/i);
      if (match && match[1]) {
        window.location.href = match[1];
        return;
      }
    }

    // Thử tìm link đích nếu có sẵn
    const link = document.querySelector('a[href*="http"]');
    if (link) {
      window.location.href = link.href;
      return;
    }

    // Nếu cần, fetch API trả link đích
    // Ví dụ giả định API: /getlink?id=...
    const btn = document.querySelector('button.red');
    if (btn) btn.click();

  } catch (err) {
    console.error('Error bypassing linkday', err);
  }
})();
